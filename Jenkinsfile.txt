pipeline {
    agent any

    stages {

        stage('Install Python Libraries') {
            steps {
                bat 'pip install pandas numpy matplotlib'
            }
        }

        stage('Run JMeter Test') {
            steps {
                bat 'jmeter -n -t test_plan.jmx -l current_results.jtl'
            }
        }

        stage('Compare Performance') {
            steps {
                bat 'python compare_runs.py baseline_results.jtl current_results.jtl'
            }
        }

    }

    post {

        always {

            archiveArtifacts artifacts: '*.png'
            archiveArtifacts artifacts: '*.html'
            archiveArtifacts artifacts: '*.csv'

        }

    }
}
